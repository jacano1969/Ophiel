145 * @returns {Boolean} Result of validation
146 */
147 function checkBlank(value) {
148     var result = false;
149     if (typeof (value) !== 'undefined') {
150         if (value.length === 0) {
151             result = true;
152         }
153     }
154     return result;

226 /**
227 * Verifies that the value is a CMIDecimal
228 *
229 * @member DataModelValidator
230 * @param {String} value (Value to be checked)
231 * @param {Boolean} score (Whether or not this is a Score variant of CMIDecimal)
232 * @returns {Boolean} Result of validation
233 */
234 function checkDecimal(value, score) {
235     var tmpDouble, pattern = /^\s*(\+|-)?((\d+(\.\d+)?)|(\.\d+))\s*$/, result = false;
236 
237     if (!(checkBlank(value))) {
238         if (this.checkPattern(value, pattern)) {
239             if (score) {
240                 tmpDouble = parseFloat(value);
241                 if (tmpDouble >= 0.0 && tmpDouble <= 100.0) {
242                     result = true;
243                 }
244                 else {
245                     debugAdd(value + ' is not a normalized score', kColor_Warning);
246                     debugAdd('0 <= ' + value + ' >= 100', kColor_Warning);
247                 }
248             }
249             else {
250                 result = true;
251             }
252         }
253         else {
254             debugAdd(value + ' is not a valid CMIDecimal Type', kColor_Warning);
255         }
256     }
257     else {
258         result = true;
259     }
260 
261     return result;
262 }

264 /**
265 * Verifies that the value is a valid vocabulary member of the provided Vocab Type
266 *
267 * @member DataModelValidator
268 * @param {String} vocabType (Vocabulary type of Element to be validated)
269 * @param {String} value (Value to be checked)
270 * @returns {Boolean} Result of validation
271 */
272 function checkVocabulary(vocabType, value) {
273     var vocabTypeList, tmpString, i, flag = false;
274 
275     debugAdd('checkVocabulary(' + vocabType + ', ' + value + ')', kColor_Info);
276 
277     if (this.vocabulary.hasOwnProperty(vocabType)) {
278         vocabTypeList = this.vocabulary[vocabType];
279 
280         for (i = 0; i < vocabTypeList.length; i++) {
281             tmpString = vocabTypeList[i];
282 
283             if (tmpString === value) {
284                 flag = true;
285                 break;
286             }
287         }
288 
289         //Check to see if the "Results" category is passing in a decimal number
290         if (vocabType === 'Result' && flag !== true) {
291             flag = this.checkDecimal(value);
292         }
293     }
294     // No match found
295     if (flag === false) {
296         debugAdd("'" + value + "' is not a valid vocabulary member for the " + vocabType + " Vocabulary Type or that Type does not exist.", kColor_Warning);
297     }
298 
299     return flag;
300 }

319 /**
320 * Verifies that the value is a CMIInteger
321 *
322 * @member DataModelValidator
323 * @param {String} value (Value to be checked)
324 * @returns {Boolean} Result of validation
325 */
326 function checkInteger(value) {
327     var flag = false;
328     if (value != null && value !== kEmptyString && !isNaN(value)) {
329         flag = (parseInt(value, 10) == value && value >= 0 && value <= 65536);
330     }
331     else {
332         debugAdd(value + ' is not an Integer', kColor_Warning);
333     }
334     return flag;
335 }

350         debugAdd(value + ' is not a Signed Integer', kColor_Warning);
351     }

422 /**
423 * Verifies that the value is a valid vocabulary member of the provided Vocab Type
424 *
425 * @member DataModelValidator
426 * @param {String} type (Type of Element to be validated)
427 * @param {String} value (Value to be checked)
428 * @param {String} vocab (Optional, can be the vocabulary type to be validated)
429 * @returns {Boolean} Result of validation
430 */
431 function Validate(type, value, vocab) {
432     var i;
433 
434     debugAdd('Validating DataType [' + type + ']...', kColor_Info);
435 
436     if (type === 'CMIVocabulary' && this.typeArray.hasOwnProperty(type)) {
437         if (eval('this.typeArray.' + type + '.call(this, vocab, value)')) {
438             return true;
439         }
440     }
441     else {
442         if (type instanceof Array) {
443             for (i = 0; i < type.length; i++) {
444                 if (this.typeArray.hasOwnProperty(type[i])) {
445                     if (eval('this.typeArray.' + type[i] + '.call(this, value)')) {
446                         return true;
447                     }
448                 }
449             }
450         }
451         else {
452             if (this.typeArray.hasOwnProperty(type)) {
453                 if (eval('this.typeArray.' + type + '.call(this, value)')) {
454                     return true;
455                 }
456             }
457         }
458     }
459 
460     return false;
461 }
