 36 /**
 37 * Adds debug output to the Web page. Checks if kDebug is true before executing.
 38 *
 39 * @param {String} debugText (What will be written to new DIV)
 40 * @param {String} debugColor (Pass a kColor_XXX constant here. Will be the DIV style's text color)
 41 * @returns {number} mNewIndex (Latest debug DIV index added)
 42 */
 43 function debugAdd(debugText, debugColor) {
 44     if (kDebug) {
 45         var mNewIndex, mNewDiv, mDivIdName, mDebugElement = document.getElementById('debug');
 46         
 47         if (typeof (debugColor) === 'undefined') {
 48             debugColor = '#FFFFFF';
 49         }
 50         mNewIndex = parseInt(mDebugElement.getAttribute("data-curindex"), 10) + 1;
 51         mNewDiv = document.createElement('div');
 52         mDivIdName = 'debug' + mNewIndex.toString() + 'Div';
 53         mDebugElement.setAttribute("data-curindex", mNewIndex.toString());
 54         mNewDiv.setAttribute('id', mDivIdName);
 55         mNewDiv.setAttribute('style', 'background-color: ' + debugColor + ';');
 56         mNewDiv.innerHTML = debugText;
 57         mDebugElement.appendChild(mNewDiv);
 58         return mNewIndex;
 59     }
 60 }

 85 /**
 86 * Sends data to the server page asynchronously.
 87 *
 88 * @param {Object} sendData (Data to be encoded and send to the server page)
 89 */
 90 function runAjax(sendData) {
 91     var urlPath;
 92 
 93     // FFffff, really, IE??
 94     if (ie) {
 95         urlPath = 'http://' + window.location.hostname + '/SCORM/server/';
 96     }
 97     else {
 98         urlPath = '../server/';
 99     }
100 
101     AjaxRequest.post({
102         'parameters': { 'data': encodeURIComponent(sendData) },
103         'url': urlPath + 'parse.php',
104         'onLoading': function () { debugAdd('Sent Data to Server...', kColor_Info); },
105         'onSuccess': function (req) {
106             if (trim(escape(req.responseText)) === 'success') {
107                 debugAdd('Commit Success response received.', kColor_Success);
108             }
109             else {
110                 debugAdd('Server reported failure during AJAX request! Message: ' + escape(req.responseText), kColor_Failure);
111             }
112         },
113         'onError': function (req) { setError('101'); debugAdd('Commit Error!\nStatusText=' + escape(req.statusText) + '\nContents=' + escape(req.responseText), kColor_Failure); }
114     });
115 }

117 /**
118 * Sends data to the server page asynchronously.
119 *
120 * @param {Number} number (Number to be padded)
121 * @param {Number} length (Length to pad to)
122 * @returns {String} str (Outputted padded number)
123 */
124 function pad(number, length) {
125     var str = kEmptyString + number;
126     while (str.length < length) {
127         str = '0' + str;
128     }
129     return str;
130 }

132 /**
133 * The LMS-managed course timer. It is used to time course if SCO does not already do so.
134 */
135 function CourseTimer() {
136     var timeDiff, startTime, endTime, seconds, minutes, hours;
137 
138     this.start = function () {
139         startTime = new Date();
140     };
141 
142     this.stop = function () {
143         endTime = new Date();
144     };
145 
146     this.clear = function () {
147         startTime = null;
148         endTime = null;
149     };
150 
151     this.buildTimeString = function () {
152         timeDiff = endTime - startTime;
153         timeDiff /= 1000;
154         seconds = parseInt(timeDiff % 60, 10);
155         timeDiff /= Math.round(60);
156         minutes = parseInt(timeDiff % 60, 10);
157         timeDiff /= Math.round(60);
158         hours = parseInt(timeDiff % 24, 10);
159         timeDiff /= Math.round(24);
160 
161         return pad(hours, 2) + ':' + pad(minutes, 2) + ':' + pad(seconds, 2);
162     };
163 }

193 /**
194 * Returns an array of tokens of the current StringTokenizer
195 *
196 * @member StringTokenizer
197 * @returns {Array} Set of tokens
198 */
199 function getTokens() {
200     var nextToken, start, end, trimmed, counter, tokens = [];
201     if (this.material.indexOf(this.separator) < 0) {
202         tokens[0] = this.material;
203         return tokens;
204     }
205     start = 0;
206     end = this.material.indexOf(this.separator, start);
207     counter = 0;
208 
209     while (this.material.length - start >= 1) {
210         nextToken = this.material.substring(start, end);
211 
212         start = end + 1;
213 
214         if (this.material.indexOf(this.separator, start + 1) < 0) {
215             end = this.material.length;
216         }
217         else {
218             end = this.material.indexOf(this.separator, start + 1);
219         }
220         trimmed = trim(nextToken);
221 
222         while (trimmed.substring(0, this.separator.length) == this.separator) {
223             trimmed = trimmed.substring(this.separator.length);
224         }
225         trimmed = trim(trimmed);
226 
227         if (trimmed == kEmptyString) {
228             continue;
229         }
230         tokens[counter] = trimmed;
231         counter++;
232     }
233     return tokens;
234 }

236 /**
237 * Returns the count of tokens in the current StringTokenizer
238 *
239 * @member StringTokenizer
240 * @returns {Number} Count of tokens
241 */
242 function countTokens() {
243     return this.tokens.length;
244 }
245 
246 /**
247 * Returns the next token in the current StringTokenizer
248 *
249 * @member StringTokenizer
250 * @returns {String} Next token in set
251 */
252 function nextToken() {
253     var returnToken;
254     if (this.tokensReturned >= this.tokens.length) {
255         return null;
256     }
257     else {
258         returnToken = this.tokens[this.tokensReturned];
259         this.tokensReturned++;
260         return returnToken;
261     }
262 }
263 
264 /**
265 * Returns whether there are more tokens in the current StringTokenizer
266 *
267 * @member StringTokenizer
268 * @returns {Boolean} True if there are more tokens
269 */
270 function hasMoreTokens() {
271     if (this.tokensReturned < this.tokens.length) {
272         return true;
273     }
274     else {
275         return false;
276     }
277 }

286     return this.tokensReturned;

289 /**
290 * Provides a data structure and set of methods for parsing through a String containing delimited tokens. Author unknown.
291 * Could be improved later by using prebuilt String member method 'split', but still should retain a lot of the nifty methods here.
292 *
293 * @class StringTokenizer Class
294 * @constructor
295 * @param {String} material (The value to be tokenized)
296 * @param {String} separator (The delimiter used in material)
297 */
298 function StringTokenizer(material, separator) {
299     this.material = material;
300     this.separator = separator;
301     this.getTokens = getTokens;
302     this.nextToken = nextToken;
303     this.countTokens = countTokens;
304     this.hasMoreTokens = hasMoreTokens;
305     this.tokensReturned = tokensReturned;
306     this.tokens = this.getTokens();
307     this.tokensReturned = 0;
308 }
