218 /**
219 * Parses an LMSGetValue() request.
220 *
221 * @member CMIRequest
222 * @param {String} theRequest (The raw request string.)
223 */
224 function parseGetRequest(theRequest) {
225     var tmpIndex, tmpIndex1, tmpInt, temp, done, stk = new StringTokenizer(theRequest, '.');
226     debugAdd('In CMIRequest::parseGetRequest', kColor_Info);
227 
228     this.totalNumberOfTokens = this.totalNumberOfTokens + stk.countTokens();
229 
230     this.model = stk.nextToken();
231     this.tokensProcessed++;
232 
233     this.pattern[0] = this.MODEL;
234 
235     this.baseCategory = stk.nextToken();
236     this.tokensProcessed++;
237 
238     this.pattern[1] = this.BASE_CATEGORY;
239 
240     if (stk.hasMoreTokens()) {
241         done = false;
242 
243         while (!done) {
244             if (this.onLastToken()) {
245                 temp = stk.nextToken();
246 
247                 if (!isNaN(temp)) {
248                     tmpInt = parseInt(temp, 10);
249                     tmpIndex = this.findNextIndexLoc();
250 
251                     this.index[tmpIndex] = tmpInt;
252 
253                     this.numOfInd++;
254 
255                     tmpIndex1 = this.findNextPatternLoc();
256 
257                     this.pattern[tmpIndex1] = this.ARRAY_INDEX;
258 
259                     this.element = this.subcategory[this.numOfSubCategories - 1];
260 
261                     tmpIndex1 = this.findNextPatternLoc();
262                     this.pattern[tmpIndex1] = this.ELEMENT;
263                 }
264                 else {
265                     this.element = temp;
266                     tmpIndex = this.findNextPatternLoc();
267 
268                     this.pattern[tmpIndex] = this.ELEMENT;
269                 }
270 
271                 this.tokensProcessed++;
272 
273                 done = true;
274             }
275             else {
276                 this.determineNextToken(stk);
277             }
278         }
279     }
280     else {
281         this.element = this.baseCategory;
282     }
283 }

285 /**
286 * Parses an LMSSetValue() request.
287 *
288 * @member CMIRequest
289 * @param {String} theRequest (The raw request string.)
290 * @param {String} theSetValue (The value to be set.)
291 */
292 function parseSetRequest(theRequest, theSetValue) {
293     var done, tmpIndex, stk = new StringTokenizer(theRequest, '.');
294     debugAdd('In CMIRequest::parseSetRequest', kColor_Info);
295 
296     this.totalNumberOfTokens = this.totalNumberOfTokens + stk.countTokens();
297 
298     this.model = stk.nextToken();
299     this.tokensProcessed++;
300 
301     this.pattern[0] = this.MODEL;
302 
303     this.baseCategory = stk.nextToken();
304     this.tokensProcessed++;
305 
306     this.pattern[1] = this.BASE_CATEGORY;
307 
308     if (stk.hasMoreTokens()) {
309         done = false;
310 
311         while (!done) {
312             if (this.onLastToken()) {
313                 this.setValue = theSetValue;
314                 this.tokensProcessed++;
315 
316                 tmpIndex = this.findNextPatternLoc();
317 
318                 this.pattern[tmpIndex] = this.VALUE;
319                 done = true;
320             }
321             else if ((this.totalNumberOfTokens - this.tokensProcessed) === 2) {
322                 this.element = stk.nextToken();
323                 this.tokensProcessed++;
324                 tmpIndex = this.findNextPatternLoc();
325 
326                 this.pattern[tmpIndex] = this.ELEMENT;
327             }
328             else {
329                 this.determineNextToken(stk);
330             }
331         }
332     }
333     else {
334         this.setValue = theSetValue;
335         this.tokensProcessed++;
336         tmpIndex = this.findNextPatternLoc();
337 
338         this.pattern[tmpIndex] = this.VALUE;
339     }
340 }

342 /**
343 * Used to determine the next token.
344 *
345 * @member CMIRequest
346 * @param {StringTokenizer} stk (String Tokenizer holding the request.)
347 */
348 function determineNextToken(stk) {
349     var temp = stk.nextToken(), tmpIndex, tmpIndex1, tmpInt;
350 
351     this.tokensProcessed++;
352 
353     if (!isNaN(temp)) {
354         tmpInt = parseInt(temp, 10);
355 
356         tmpIndex = this.findNextIndexLoc();
357 
358         this.index[tmpIndex] = tmpInt;
359 
360         this.numOfInd++;
361 
362         tmpIndex1 = this.findNextPatternLoc();
363 
364         this.pattern[tmpIndex1] = this.ARRAY_INDEX;
365     }
366     else {
367         tmpIndex = this.findNextSubLoc();
368 
369         this.subcategory[tmpIndex] = temp;
370 
371         this.numOfSubCategories++;
372 
373         tmpIndex1 = this.findNextPatternLoc();
374 
375         this.pattern[tmpIndex1] = this.SUB_CATEGORY;
376     }
377 }

379 /**
380 * Determines if on last token.
381 *
382 * @member CMIRequest
383 * @returns {Boolean} True if on last token.
384 */
385 function onLastToken() {
386     var flag = false;
387 
388     if ((this.totalNumberOfTokens - this.tokensProcessed) === 1) {
389         flag = true;
390     }
391     return flag;
392 }

394 /**
395 * Uses pattern collection to determine the next available token.
396 *
397 * @member CMIRequest
398 * @returns {String} Next available token.
399 */
400 function getNextToken() {
401     var tmpIndex, tmpPattern, tmpPat, rtrnString = kEmptyString;
402 
403     if (this.pattern[this.pattern_counter] != kEmptyString) {
404         tmpPat = this.pattern[this.pattern_counter];
405         tmpPattern = parseInt(tmpPat, 10);
406 
407         if (tmpPattern == this.SUB_CATEGORY) {
408             if (this.subcategory[this.numOfSubCatReturned] != kEmptyString) {
409                 rtrnString = this.subcategory[this.numOfSubCatReturned];
410                 this.numOfSubCatReturned++;
411             }
412             else {
413                 debugAdd('Index Out of Bounds', kColor_Warning);
414             }
415         }
416         else if (tmpPattern == this.ARRAY_INDEX) {
417             if (this.index[this.numOfIndReturned] != kEmptyString) {
418                 tmpIndex = this.index[this.numOfIndReturned];
419                 this.numOfIndReturned++;
420                 rtrnString = tmpIndex.toString();
421             }
422             else {
423                 debugAdd('Index Out of Bounds', kColor_Warning);
424             }
425         }
426         else if (tmpPattern == this.ELEMENT) {
427             rtrnString = this.element;
428 
429             this.resetPatternCounter();
430         }
431         else if (tmpPattern == this.VALUE) {
432             rtrnString = this.setValue;
433         }
434 
435         if (tmpPattern != this.ELEMENT) {
436             this.incrementPatternCount();
437         }
438         this.tokensRequested++;
439     }
440     else {
441         debugAdd('Index Out of Bounds', kColor_Warning);
442     }
443 
444     return rtrnString;
445 }

447 /**
448 * Determines if there are more tokens to process.
449 *
450 * @member CMIRequest
451 * @returns {Boolean} True if there are more tokens.
452 */
453 function hasMoreTokensToProcess() {
454     var flag = true;
455 
456     // If dealing with LMSSetValue() request don't count the setValue
457     if (this.IsSetRequest) {
458         if (((this.totalNumberOfTokens - 1) - this.tokensRequested) == 0) {
459             flag = false;
460         }
461     }
462     else {
463         if ((this.totalNumberOfTokens - this.tokensRequested) == 0) {
464             flag = false;
465         }
466     }
467 
468     return flag;
469 }

471 /**
472 * Invoked when processing of a request is complete.
473 *
474 * @member CMIRequest
475 */
476 function done() {
477     this.resetPatternCounter();
478     this.tokensRequested = 1;
479     this.numOfSubCatReturned = 0;
480     this.numOfIndReturned = 0;
481     return;
482 }

484 /**
485 * Determines next available position in pattern array.
486 *
487 * @member CMIRequest
488 * @returns {Number} Next available position.
489 */
490 function findNextPatternLoc() {
491     var i, result = -1, value = -1;
492 
493     for (i = 0; i <= this.numPat; i++) {
494         if (value == parseInt(this.pattern[i], 10)) {
495             result = i;
496             break;
497         }
498     }
499 
500     return result;
501 }
502 
503 /**
504 * Determines next available position in Index array.
505 *
506 * @member CMIRequest
507 * @returns {Number} Next available position.
508 */
509 function findNextIndexLoc() {
510     var i, result = -1, value = -1;
511 
512     for (i = 0; i <= this.numInd; i++) {
513         if (value == parseInt(this.index[i], 10)) {
514             result = i;
515             break;
516         }
517     }

522 /**
523 * Determines next available position in subcategory array.
524 *
525 * @member CMIRequest
526 * @returns {Number} Next available position.
527 */
528 function findNextSubLoc() {
529     var i, result = -1, tmpString = '-1';
530 
531     for (i = 0; i <= this.numSub; i++) {
532         if (this.subcategory[i] == tmpString) {
533             result = i;
534             break;
535         }
536     }
537 
538     return result;
539 }

541 /**
542 * Generates debug output based on the current CMIRequest object.
543 *
544 * @member CMIRequest
545 * @returns {Number} Next available position.
546 */
547 function showRequest() {
548     var i;
549     debugAdd('Base Category: ' + this.baseCategory, kColor_Info);
550     debugAdd('Element: ' + this.element, kColor_Info);
551     debugAdd('Model: ' + this.model, kColor_Info);
552     debugAdd('Number of Indices: ' + this.numOfInd, kColor_Info);
553     debugAdd('Number of SubCategories: ' + this.numOfSubCategories, kColor_Info);
554 
555     for (i = 0; i < this.numOfSubCategories; i++) {
556         debugAdd('Subcategory[' + i + ']: ' + this.getSubCategory(i), kColor_Info);
557     }
558 
559     for (i = 0; i < this.numOfInd; i++) {
560         debugAdd('Index[' + i + ']: ' + this.index[i], kColor_Info);
561     }
562 
563     for (i = 0; i < this.numPat; i++) {
564         if (parseInt(this.pattern[i], 10) > -1) {
565             debugAdd('Pattern[' + i + ']: ' + this.pattern[i], kColor_Info);
566         }
567     }
568 
569     if (this.isForASetRequest()) {
570         debugAdd('Set Value: ' + this.setValue, kColor_Info);
571     }
572 }

574 /**
575 * Defines the CMIRequest object, which is fed as part of getting/setting an actual data value.
576 *
577 * @class CMIRequest Object
578 * @constructor
579 * @param {String} theRequest (The request and its various parameters. (example, cmi.core.student_id))
580 * @param {Boolean} getRequestFlag (True if this is for an LMSGetValue())
581 */
582 function CMIRequest(theRequest, getRequestFlag) {
583     var i, setTok, numTokensForSet, numTokensProcessed, theSetValue = kEmptyString, theNewRequest = kEmptyString;
584 
585     if (getRequestFlag) {
586         debugAdd('Building CMIRequest for a LMSGetValue(' + theRequest + ')', kColor_Info);
587     }
588     else {
589         debugAdd('Building CMIRequest for a LMSSetValue(' + theRequest + ')', kColor_Info);
590     }
591 
592     if (typeof (theRequest) !== 'undefined') {
593         this.request = theRequest;
594     }
595     else {
596         this.request = kEmptyString;
597     }
598 
599     if (typeof (getRequestFlag) !== 'undefined') {
600         this.IsGetRequest = getRequestFlag;
601     }
602     else {
603         this.IsGetRequest = false;
604     }
605 
606     this.model = kEmptyString;
607 
608     this.baseCategory = kEmptyString;
609 
610     this.numSub = 10;
611     this.subcategory = [];
612 
613     this.numOfSubCategories = 0;
614 
615     this.numOfSubCatReturned = 0;
616 
617     this.numInd = 10;
618     this.index = [];
619 
620     this.numOfInd = 0;
621 
622     this.numOfIndReturned = 0;
623 
624     this.element = kEmptyString;
625 
626     this.setValue = kEmptyString;
627 
628     this.totalNumberOfTokens = 0;
629 
630     this.tokensRequested = 1;
631 
632     this.tokensProcessed = 0;
633 
634     this.IsSetRequest = false;
635 
636     // 1 - model
637     // 2 - base category
638     // 3 - sub category
639     // 4 - array index
640     // 5 - element
641     // 6 - value to use for set
642     // cmi.core.student_id -- 1,2,5
643     // cmi.student_data.tries.n.score.raw -- 1,2,3,4,3,5
644     this.numPat = 10;
645     this.pattern = [];
646 
647     this.MODEL = 1;
648     this.BASE_CATEGORY = 2;
649     this.SUB_CATEGORY = 3;
650     this.ARRAY_INDEX = 4;
651     this.ELEMENT = 5;
652     this.VALUE = 6;
653 
654     // we're starting at 2!
655     this.pattern_counter = 2;
656 
657     this.index = new Array(this.numInd);
658 
659     for (i = 0; i < this.index.length; i++) {
660         this.index[i] = -1;
661     }
662 
663     this.pattern = new Array(this.numPat);
664 
665     for (i = 0; i < this.pattern.length; i++) {
666         this.pattern[i] = -1;
667     }
668 
669     this.subcategory = new Array(this.numSub);
670 
671     for (i = 0; i < this.subcategory.length; i++) {
672         this.subcategory[i] = '-1';
673     }
674 
675     if (this.IsGetRequest) {
676         this.IsSetRequest = false;
677         this.parseGetRequest(theRequest);
678     }
679     else {
680         this.IsSetRequest = true;
681 
682         setTok = new StringTokenizer(theRequest, ',');
683         numTokensForSet = setTok.countTokens();
684         numTokensProcessed = 0;
685 
686         theNewRequest = setTok.nextToken();
687         numTokensProcessed++;
688 
689         debugAdd('Request: [' + theNewRequest + ']', kColor_Info);
690 
691         if (numTokensForSet === 1) {
692             theSetValue = kEmptyString;
693         }
694         else {
695             while (numTokensProcessed < numTokensForSet) {
696                 theSetValue += setTok.nextToken();
697                 numTokensProcessed++;
698 
699                 if (numTokensProcessed != numTokensForSet) {
700                     theSetValue += ',';
701                 }
702             }
703 
704             debugAdd('Set Value: [' + theSetValue + ']', kColor_Info);
705         }
706 
707         this.totalNumberOfTokens++;
708         this.parseSetRequest(theNewRequest, theSetValue);
709     }
710 
711     this.showRequest();
712 }
