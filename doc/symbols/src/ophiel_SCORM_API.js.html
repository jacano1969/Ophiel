 19 /**
 20 * Initializes the LMS
 21 *
 22 * @member apiClass
 23 * @param {String} parameter (must be empty String)
 24 * @returns {CMIBoolean} ('false' - failed. 'true' - succeeded.)
 25 */
 26 function LMSInitialize(parameter) {
 27     setError('0');
 28     debugAdd('init LMS...', kColor_Info);
 29     var mInitResult = kCMIBooleanFalse;
 30 
 31     if (!(parameter === kEmptyString || parameter === 'null')) {
 32         setError('201');
 33         debugAdd('Code 201', kColor_Warning);
 34         return mInitResult;
 35     }
 36 
 37     if (this.isLMSInitialized === true) {
 38         setError('101');
 39         debugAdd('Code 101', kColor_Warning);
 40     } else {
 41         Session.clear();
 42         setError('0');
 43         courseTime.start();
 44         debugAdd('Began recording LMS Course Time.', kColor_Info);
 45         this.isLMSInitialized = true;
 46         mInitResult = kCMIBooleanTrue;
 47     }
 48 
 49     return mInitResult;
 50 }

 52 /**
 53 * Retrieves an LMS value.
 54 *
 55 * @member apiClass
 56 * @param {String} element
 57 * @param {Boolean} ignoreFlag (Hack the Gibson- ignore the element's read-only status.)
 58 * @returns {String}
 59 */
 60 function LMSGetValue(element, ignoreFlag) {
 61     var mDataManager, mCMIRequestInstance;
 62     setError('0');
 63     mDataManager = new DataManager();
 64     mCMIRequestInstance = new CMIRequest(element, true);
 65     return mDataManager.getValue(mCMIRequestInstance, ignoreFlag);
 66 }

 68 /**
 69 * Sets an LMS value.
 70 *
 71 * @member apiClass
 72 * @param {String} element
 73 * @param {String} value
 74 * @returns {CMIBoolean} ('false' - failed. 'true' - succeeded.)
 75 */
 76 function LMSSetValue(element, value) {
 77     var mDataManager, mCMIRequestInstance;
 78     setError('0');
 79     if (typeof (value) !== 'undefined') {
 80         element = element + ',' + value;
 81     }
 82     mDataManager = new DataManager();
 83     mCMIRequestInstance = new CMIRequest(element, false);
 84 
 85     return mDataManager.setValue(mCMIRequestInstance);
 86 }

 88 /**
 89 * Commits LMS state.
 90 *
 91 * @member apiClass
 92 * @param {String} parameter (must be empty String)
 93 * @returns {CMIBoolean} ('false' - failed. 'true' - succeeded.)
 94 */
 95 function LMSCommit(parameter) {
 96     var courseTimeString, mResult = kCMIBooleanFalse;
 97     setError('0');
 98 
 99     if (!(parameter === kEmptyString || parameter === 'null')) {
100         setError('201');
101         debugAdd('Commit failure! Code 201', kColor_Failure);
102         return mResult;
103     }
104 
105     debugAdd('Committing LMS state...', kColor_Info);
106     runAjax(Session.get('CMIModel'));
107     mResult = kCMIBooleanTrue;
108 
109     return mResult;
110 }

112 /**
113 * Checks if session_time has been set, and if not, retrieves LMS's timer and sets it.
114 *
115 * @member apiClass
116 * @returns {Boolean} (Indicates if the LMS set the value)
117 */
118 function CheckTime() {
119     var courseTimeString, mResult = false;
120 
121     courseTime.stop();
122     courseTimeString = courseTime.buildTimeString();
123     debugAdd('Checking time. LMS-recorded Course Time: ' + courseTimeString, kColor_Info);
124     if (this.LMSGetValue('cmi.core.session_time', true) === kEmptyString) {
125         this.LMSSetValue('cmi.core.session_time', courseTimeString);
126         mResult = true;
127         debugAdd('The LMS set the course time manually because the SCO did not! For shame.', kColor_Warning);
128     }
129 
130     return mResult;
131 }
132 
133 /**
134 * Turns off communication with the SCO.
135 *
136 * @member apiClass
137 * @param {String} parameter (must be empty String)
138 * @returns {CMIBoolean} ('false' - failed. 'true' - succeeded.)
139 */
140 function LMSFinish(parameter) {
141     var mResult = kCMIBooleanFalse;
142     setError('0');
143 
144     if (!(parameter === kEmptyString || parameter === 'null')) {
145         setError('201');
146         debugAdd('Code 201', kColor_Warning);
147         return mResult;
148     }
149 
150     debugAdd('Finishing LMS/SCO communication...', kColor_Info);
151     if (this.CheckInitialization()) {
152         this.CheckTime();
153         this.LMSCommit("");
154         this.isLMSInitialized = false;
155         mResult = kCMIBooleanTrue;
156         debugAdd('LMS is now finished.', kColor_Success);
157         return mResult;
158     }
159     else {
160         debugAdd('LMS is already finished!', kColor_Failure);
161         return mResult;
162     }
163 }

176 /**
177 * Retrieves a description of the error represented by the error code passed.
178 *
179 * @member apiClass
180 * @param {String} errorNumber
181 * @returns {String}
182 */
183 function LMSGetErrorString(errorNumber) {
184     debugAdd('Retrieving error description...', kColor_Info);
185     var errorString;
186     if (errorNumber !== kEmptyString) {
187         errorString = kErrorStringArray[errorNumber];
188         if (typeof (errorString) === 'undefined') {
189             errorString = kEmptyString;
190         }
191     } else {
192         errorString = kEmptyString;
193     }
194 
195     return errorString;
196 }

209 /**
210 * Checks and returns whether or not the LMS is initialized, and if not, triggers the error handler.
211 *
212 * @member apiClass
213 * @returns {Boolean}
214 */
215 function CheckInitialization() {
216     setError('0');
217     if (this.isLMSInitialized !== true) {
218         setError('301');
219         debugAdd('Code 301', kColor_Warning);
220     }
221     return this.isLMSInitialized;
